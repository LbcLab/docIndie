175 static inline bool addons_uuid_read( const char *psz_uuid, addon_uuid_t *p_uuid )
176 {
177  if ( !psz_uuid ) return false;
178  if ( strlen( psz_uuid ) < ADDON_UUID_PSZ_SIZE ) return false;
179 
180  int i = 0, j = 0;
181  while ( i<ADDON_UUID_PSZ_SIZE )
182  {
183  if ( *( psz_uuid + i ) == '-' )
184  i++;
185  int v;
186  sscanf( psz_uuid + i, "%02x", &v );
187  (*p_uuid)[j++] = v & 0xFF;
188  i+=2;
189  }
190 
191  return true;
192 }

194 static inline char * addons_uuid_to_psz( const addon_uuid_t * p_uuid )
195 {
196  char *psz = (char*) calloc( ADDON_UUID_PSZ_SIZE + 1 , sizeof(char) );
197  if ( psz )
198  {
199  int i=0;
200  char *p = psz;
201  while ( i < ADDON_UUID_SIZE )
202  {
203  if ( i == 4 || i== 7 || i== 9 || i== 11 )
204  *p++ = '-';
205  int v = 0xFF & (*p_uuid)[i];
206  sprintf( p, "%02x", v );
207  p += 2;
208  i++;
209  }
210  }
211  return psz;
212 }
